maps-quiz3: What would the following code print?:

function mapChanger(someMap){
  if (someMap.has("a")) {
    someMap.set("b", someMap.get("a"));
  }
  someMap.delete("c");
  return someMap;
}

let myMap = new Map();
myMap.set("a", "happy");
myMap.set("b", "sad");
myMap.set("c", "excited");

myMap = mapChanger(myMap);
console.log(myMap.get("b"));
